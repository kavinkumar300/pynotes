class Stack:
    def __init__(self):
        self.stack = []

def push(self, item):
        print(f"Pushed {item}")
        self.stack.append(item)

 def pop(self):
        if not self.is_empty():
            print(f"Popped {self.stack[-1]}")
            return self.stack.pop()
        return None

def is_empty(self):
        return len(self.stack) == 0

 def peek(self):
        return self.stack[-1] if self.stack else None

def display(self):
        print("Stack:", self.stack)

class Queue:
    def __init__(self):
        self.queue = []

 def enqueue(self, item):
        print(f"Enqueued {item}")
        self.queue.append(item)

 def dequeue(self):
        if not self.is_empty():
            print(f"Dequeued {self.queue[0]}")
            return self.queue.pop(0)
        return None

 def is_empty(self):
        return len(self.queue) == 0

  def display(self):
        print("Queue:", self.queue)

class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LinkedList:
    def __init__(self):
        self.head = None

def insert_end(self, data):
        new_node = Node(data)
        if not self.head:
            self.head = new_node
            print(f"Inserted {data} at head")

 else:
            temp = self.head
            while temp.next:
                temp = temp.next
            temp.next = new_node
            print(f"Inserted {data} at end")

  def display(self):
        temp = self.head
        print("LinkedList:", end=" ")
        while temp:
            print(temp.data, end=" -> ")
            temp = temp.next
        print("None")

class TreeNode:
    def __init__(self, val):
        self.val = val
        self.left = None
        self.right = None

def inorder(root):
    if root:
        inorder(root.left)
        print(root.val, end=" ")
        inorder(root.right)

def preorder(root):
    if root:
        print(root.val, end=" ")
        preorder(root.left)
        preorder(root.right)

def postorder(root):
    if root:
        postorder(root.left)
        postorder(root.right)
        print(root.val, end=" ")

class BST:
    def __init__(self):
        self.root = None

 def insert(self, val):
        self.root = self._insert(self.root, val)

def _insert(self, node, val):
        if not node:
            print(f"Inserted {val}")
            return TreeNode(val)
        if val < node.val:
            node.left = self._insert(node.left, val)
        else:
            node.right = self._insert(node.right, val)
        return node

    def inorder(self):
        print("Inorder Traversal:")
        inorder(self.root)
        print()
