print("\n=== Bubble Sort ===")
arr = [64, 34, 25, 12, 22, 11, 90]

=== Bubble Sort ===

print("Unsorted:", arr)
n = len(arr)
for i in range(n):
    for j in range(0, n - i - 1):
        if arr[j] > arr[j + 1]:
            arr[j], arr[j + 1] = arr[j + 1], arr[j]

Unsorted: [64, 34, 25, 12, 22, 11, 90]

print("Sorted:", arr)

Sorted: [11, 12, 22, 25, 34, 64, 90]

print("\n=== Binary Search ===")

=== Binary Search ===

arr = [2, 4, 6, 8, 10, 12, 14]
target = 10
low = 0
high = len(arr) - 1
found = False

while low <= high:
    mid = (low + high) // 2
    if arr[mid] == target:
        print(f"Found {target} at index {mid}")
        found = True
        break

elif arr[mid] < target:
        low = mid + 1

else:
        high = mid - 1

if not found:
    print(f"{target} not found")

print("\n=== Factorial Using Recursion ===")
def factorial(n):
    if n == 1:
        return 1
    return n * factorial(n - 1)

print("5! =", factorial(5))

head = Node(1)
second = Node(2)

while current:
