def get_student_details():
    name = input("Enter student name: ")
    roll_no = input("Enter roll number: ")
    return name, roll_no

def get_marks(subjects):
    marks = {}
    for subject in subjects:

while True:
            try:
                mark = float(input(f"Enter marks for {subject}: "))

if 0 <= mark <= 100:
                    marks[subject] = mark
                    break

 else:
                    print("Marks should be between 0 and 100.")

except ValueError:
                print("Enter a valid number.")
    return marks

def calculate_average(marks):
    total = sum(marks.values())
    return total / len(marks)

def get_remark(average):
    if average >= 90:
        return "Excellent"

elif average >= 75:
        return "Very Good"

elif average >= 60:
        return "Good"

elif average >= 40:
        return "Pass"

else:
        return "Fail"

def display_report(name, roll_no, marks, average, remark):

print("\n--- Student Report ---")
    print(f"Name      : {name}")
    print(f"Roll No   : {roll_no}")

 print("\nMarks:")
    for subject, mark in marks.items():
        print(f"{subject:10}: {mark}")

 print(f"\nAverage   : {average:.2f}")
    print(f"Remark    : {remark}")

def main():
    subjects = ["Math", "Science", "English", "History", "Computer"]
    name, roll_no = get_student_details()
    marks = get_marks(subjects)
    average = calculate_average(marks)
    remark = get_remark(average)
    display_report(name, roll_no, marks, average, remark)

if __name__ == "__main__":
    main()
